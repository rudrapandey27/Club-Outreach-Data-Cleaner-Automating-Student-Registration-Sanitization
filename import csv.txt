import csv

def is_valid_registration(reg_num):
    """
    Checks if the registration number matches the university format.
    Example valid format: 25BEC10025 (10 characters, starts with year, alphanumeric).
    """
    reg_num = reg_num.strip().upper()
    if len(reg_num) == 10 and reg_num[:2].isdigit() and reg_num.isalnum():
        return True
    return False

def is_valid_email(email):
    """
    Performs a basic check to ensure the email has an '@' and a domain.
    """
    email = email.strip()
    if "@" in email and "." in email.split("@")[-1]:
        return True
    return False

def process_outreach_data(input_filename, clean_filename, error_filename):
    """
    Reads raw club sign-up data, removes duplicates, and separates 
    valid entries from invalid ones.
    """
    clean_data = []
    error_data = []
    seen_registration_numbers = set() # Using a Set is highly efficient for checking duplicates

    try:
        with open(input_filename, mode='r', encoding='utf-8') as file:
            reader = csv.reader(file)
            header = next(reader) # Skip the header row
            
            for row in reader:
                # Assuming CSV format: Name, Registration Number, Email
                if len(row) != 3:
                    error_data.append(row + ["Missing data fields"])
                    continue
                
                name, reg_num, email = row
                reg_num = reg_num.strip().upper()

                # 1. Check for duplicates
                if reg_num in seen_registration_numbers:
                    error_data.append(row + ["Duplicate Entry"])
                    continue
                
                # 2. Validate format
                if not is_valid_registration(reg_num):
                    error_data.append(row + ["Invalid Registration Format"])
                    continue
                
                if not is_valid_email(email):
                    error_data.append(row + ["Invalid Email Format"])
                    continue

                # If it passes all checks, it's clean data
                seen_registration_numbers.add(reg_num)
                clean_data.append([name.strip(), reg_num, email.strip()])

        # Export the clean data
        with open(clean_filename, mode='w', newline='', encoding='utf-8') as clean_file:
            writer = csv.writer(clean_file)
            writer.writerow(["Name", "Registration Number", "Email"])
            writer.writerows(clean_data)

        # Export the error log
        with open(error_filename, mode='w', newline='', encoding='utf-8') as error_file:
            writer = csv.writer(error_file)
            writer.writerow(["Name", "Registration Number", "Email", "Error Reason"])
            writer.writerows(error_data)

        print(f"Success! Processed {len(clean_data) + len(error_data)} records.")
        print(f"Clean entries saved to: {clean_filename}")
        print(f"Errors saved to: {error_filename}")

    except FileNotFoundError:
        print(f"Error: Could not find the file '{input_filename}'.")

# Run the program
if __name__ == "__main__":
    print("Starting Fusion Club Data Cleaner...")
    process_outreach_data("raw_signups.csv", "clean_attendees.csv", "error_log.csv")
